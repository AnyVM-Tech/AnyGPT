from openai import OpenAI

client = OpenAI(
    base_url="https://gpt.anyvm.tech/v1",
    api_key="your-anygpt-api-key"
)

response = client.chat.completions.create(
    model="gpt-4o",
    messages=[
        {"role": "system", "content": "Be concise."},
        {"role": "user", "content": "Summarize our launch plan."}
    ]
)
